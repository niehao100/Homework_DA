function m_note_1(note,time,base,f)
    %杈撳叆鍙傛暟
    %note闊崇,
	%-10
	%琛ㄧず绌洪煶
    %time鎸佺画鏃堕棿
    %base鍩鸿皟
    %f閲囨牱棰戠巼
    n=[-5:19];
    freq=base*2.^kron(1/12,n);
    t=[0:time*f]/f;
    if note==10
        s=0*t;
    else
        s=cos(2*pi*freq(note+6)*t);
        sound(s,f);
        pause(time+0.01);
    end
    return
